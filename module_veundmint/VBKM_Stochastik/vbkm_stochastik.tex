\begin{MExercise}

\begin{MExerciseItems}
\item{\MEquationItem{$\MDVec{P Q}$}{\MFunctionQuestion{15}{(-2,-2,-3)}{5}{x}{5}}.}
\item{\MEquationItem{$\MDVec{R Q}$}{\MFunctionQuestion{15}{(4,-2,-2)}{5}{x}{5}}.}
\end{MExerciseItems}

Bestimmen Sie den Mittelpunkt dieser Vektoren: \MEquationItem{$\MVec{m}$}{\MFunctionQuestion{15}{(1,1,1)}{5}{x}{5}}.

\begin{MExercise}
Finden Sie den Schnittpunkt der beiden in Punkt-Richtungsform gegebenen Geraden
$$
\MVec{x} \;=\; \MVector{1\\1\\2}+\alpha\cdot \MVector{1\\1\\1}\;\; \text{und}\;\;
\MVec{x} \;=\; \MVector{0\\1\\-2}+\beta\cdot \MVector{-2\\-4\\4}\: .
$$
\begin{MExerciseItems}
\item{Der Schnittpunkt (als Vektor) ist \MFunctionQuestion{15}{(-1,-1,0)}{5}{x}{5}.}
\item{Man erh\"alt ihn als Punkt der ersten Geraden f\"ur den Parameter \MEquationItem{$\alpha$}{\MParsedQuestion{7}{-2}{3}}.}
\item{Man erh\"alt ihn als Punkt der zweiten Geraden f\"ur den Parameter \MEquationItem{$\beta$}{\MParsedQuestion{7}{1/2}{3}}.}
\end{MExerciseItems}
\end{MExercise}

\end{MTest}
