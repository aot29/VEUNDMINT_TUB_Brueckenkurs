  fkt(\x) = \x/2 * sqrt((\x + 3)*(4 - \x));
}
] %[every node/.style={fill=white}]

 node[above] {$\x$};

 node[below] {$\x$};

 node[right] {$\y$};
\foreach \y in {1, 2, 3} \draw (0, \y) -- ++(-0.1, 0) %
 node[left] {$\y$};
%\node[below left] at (0, 0) {$0$};
%Funktion:
\draw[domain=-3:4,samples=120,color=\jccolorfkt] %
 plot (\x, {fkt(\x)});
